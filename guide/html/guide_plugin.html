[data-modules="customScroll"]

jQuery custom content scroller

URL : http://manos.malihu.gr/jquery-custom-content-scroller/




추가 요소

// this – 스크롤바를 포함하는 원본 요소
// this.mcs.content – jquery 객체로서의 원래 내용 래퍼
// this.mcs.top – 콘텐츠의 최상위 위치 (픽셀)
// this.mcs.left – 컨텐츠의 왼쪽 위치 (픽셀)
// this.mcs.draggerTop – 스크롤바 드래 거의 상단 위치 (픽셀)
// this.mcs.draggerLeft – 스크롤바 드래 거의 왼쪽 위치 (픽셀)
// this.mcs.topPct – 컨텐츠 수직 스크롤 백분율
// this.mcs.leftPct – 콘텐츠 가로 스크롤 비율
// this.mcs.direction – 컨텐츠의 스크롤 방향 (y 또는 x)
$('[data-modules="customScroll"]').on('scrollStart', function(e, el){
    // 스크롤 시작시 이벤트     console.log('start')
    console.log(this.mcs.topPct)
}).on('scrolling', function(e, el){
    // 스크롤 중...     console.log(this.mcs.topPct)
}).on('scrollEnd', function(e, el){
    // 스크롤 종료시 이벤트     console.log('end');
    console.log(this.mcs)
    console.log(this.mcs.topPct)
})
