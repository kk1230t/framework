Intro


Custom $(window) event 에 대한 내용입니다.

"changeSize" Event

UI.DEFAULT_SCREEN_SIZE 에서 지정한 해상도별 mode가 변경시 발생 


	// 기본 해상도 분기
	var DEFAULT_SCREEN_SIZE = [
		{
			mode: 'mobile',
			min: 0,
			max: 767
		},
		{
			mode: 'tablet',
			min: 768,
			max: 1024
		},
		{
			mode: 'web',
			min: 1025,
			max: 3840
		}
	];


	/**
	* @event 	이벤트 객체
	* @mode 	현재 mode
	*/
	$(window).on('changeSize', function(event, mode){
		console.log(mode);
	})
